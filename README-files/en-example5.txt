function kings(start_bid)
	?$(start_bid) no kings
	$(start_bid + 1) king of ♣ or ♦+♥
	$(start_bid + 2) king of ♦ or ♣+♥
	$(start_bid + 3) king of ♥ or ♣+♦
	$(start_bid + 4) kings of ♣+♦+♥
end
